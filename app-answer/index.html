<script>

var answer = {
    uid:'10000', // 用户UID(暂定是否需要)
    uuid:'abcdefg01abcdefg01', // 题目的UUID,唯一标识
    type:1,     // 题目类型 1:单选，2:多选 .....(类型暂定，需要后端配合定义)

    // answers:['ABC'] // 多选题答案。由于多选题虽然答案有多个，但是，看作是一个待答区
}

</script>
